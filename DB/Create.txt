
CREATE DATABASE sistema_academico;
USE sistema_academico;

SET FOREIGN_KEY_CHECKS = 0;

-- =========================
-- ADMINISTRACIÓN
-- =========================

CREATE TABLE persona (
    id_persona INT AUTO_INCREMENT PRIMARY KEY,
    tipo_documento_identidad ENUM('CC','TI','CE','PA') NOT NULL DEFAULT 'CC',
    numero_documento_identidad VARCHAR(20) NOT NULL UNIQUE,
    nombres VARCHAR(50) NOT NULL,
    apellidos VARCHAR(50) NOT NULL,
    fecha_nacimiento DATE NOT NULL,
    estado_civil ENUM('Soltero','Casado','Union Libre','Divorciado','Viudo'),
    pais_residencia VARCHAR(100),
    departamento_residencia VARCHAR(100),
    ciudad_residencia VARCHAR(100),
    direccion TEXT,
    telefono VARCHAR(20),
    email VARCHAR(100),
    tipo_sangre ENUM('A+','A-','B+','B-','O+','O-','AB+','AB-')
) ENGINE=InnoDB;
-- Añadimos la columna estado a persona (que es la que pide el error 1054)
ALTER TABLE persona ADD COLUMN estado ENUM('activo', 'inactivo') DEFAULT 'activo';

CREATE TABLE usuario (
    id_persona INT PRIMARY KEY,
    login VARCHAR(50) NOT NULL UNIQUE,
    password VARCHAR(100) NOT NULL,
    estado ENUM('Activo','Inactivo') DEFAULT 'Activo',
    FOREIGN KEY (id_persona) REFERENCES persona(id_persona) ON DELETE CASCADE
) ENGINE=InnoDB;

CREATE TABLE rol (
    id_rol INT AUTO_INCREMENT PRIMARY KEY,
    nombre_rol VARCHAR(50) NOT NULL UNIQUE,
    descripcion TEXT
) ENGINE=InnoDB;

CREATE TABLE usuario_rol (
    id_persona INT NOT NULL,
    id_rol INT NOT NULL,
    fecha_asignacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (id_persona, id_rol),
    FOREIGN KEY (id_persona) REFERENCES usuario(id_persona) ON DELETE CASCADE,
    FOREIGN KEY (id_rol) REFERENCES rol(id_rol) ON DELETE CASCADE
) ENGINE=InnoDB;

CREATE TABLE modulo (
    id_modulo INT AUTO_INCREMENT PRIMARY KEY,
    nombre_corto_modulo VARCHAR(20) NOT NULL UNIQUE,
    nombre_modulo VARCHAR(100),
    descripcion_modulo TEXT
) ENGINE=InnoDB;

CREATE TABLE funcionalidad (
    id_funcionalidad INT AUTO_INCREMENT PRIMARY KEY,
    nombre_corto_funcionalidad VARCHAR(20) NOT NULL UNIQUE,
    nombre_funcionalidad VARCHAR(100),
    descripcion_funcionalidad TEXT,
    id_modulo INT NOT NULL,
    ruta VARCHAR(50),
    FOREIGN KEY (id_modulo) REFERENCES modulo(id_modulo) ON DELETE CASCADE
) ENGINE=InnoDB;

CREATE TABLE rol_funcionalidad (
    id_rol INT NOT NULL,
    id_funcionalidad INT NOT NULL,
    PRIMARY KEY (id_rol, id_funcionalidad),
    FOREIGN KEY (id_rol) REFERENCES rol(id_rol) ON DELETE CASCADE,
    FOREIGN KEY (id_funcionalidad) REFERENCES funcionalidad(id_funcionalidad) ON DELETE CASCADE
) ENGINE=InnoDB;

-- =========================
-- PERSONAS
-- =========================

CREATE TABLE estudiante (
    id_persona INT PRIMARY KEY,
    fecha_ingreso DATE,
    alergias TEXT,
    condicion_especial TEXT DEFAULT NULL,
    observaciones_medicas TEXT,
    FOREIGN KEY (id_persona) REFERENCES persona(id_persona) ON DELETE CASCADE
) ENGINE=InnoDB;

CREATE TABLE docente (
    id_persona INT PRIMARY KEY,
    ultimo_titulo_profesional VARCHAR(150),
    actual_cargo VARCHAR(100),
    fecha_contratacion DATE,
    FOREIGN KEY (id_persona) REFERENCES persona(id_persona) ON DELETE CASCADE
) ENGINE=InnoDB;

-- =========================
-- ESTRUCTURA ACADÉMICA
-- =========================

CREATE TABLE nivel (
    id_nivel INT PRIMARY KEY,
    nombre_nivel VARCHAR(100) NOT NULL,
    descripcion_nivel TEXT
) ENGINE=InnoDB;

CREATE TABLE grado (
    id_grado INT PRIMARY KEY,
    id_nivel INT NOT NULL,
    nombre_grado VARCHAR(50) NOT NULL,
    descripcion_grado TEXT,
    FOREIGN KEY (id_nivel) REFERENCES nivel(id_nivel)
) ENGINE=InnoDB;

CREATE TABLE area (
    id_area INT AUTO_INCREMENT PRIMARY KEY,
    id_grado INT NOT NULL,
    nombre_area VARCHAR(100) NOT NULL,
    descripcion_area TEXT,
    FOREIGN KEY (id_grado) REFERENCES grado(id_grado)
) ENGINE=InnoDB;

CREATE TABLE asignatura (
    id_asignatura INT PRIMARY KEY,
    id_area INT NOT NULL,
    nombre_asignatura VARCHAR(100) NOT NULL,
    descripcion_asignatura TEXT,
    ihs INT,
    FOREIGN KEY (id_area) REFERENCES area(id_area)
) ENGINE=InnoDB;

CREATE TABLE desempenio (
    id_desempenio INT AUTO_INCREMENT PRIMARY KEY,
    id_asignatura INT NOT NULL,
    descripcion_desempenio TEXT NOT NULL,
    FOREIGN KEY (id_asignatura) REFERENCES asignatura(id_asignatura)
) ENGINE=InnoDB;

-- =========================
-- AÑO, PERIODOS, GRUPOS
-- =========================

CREATE TABLE anio_escolar (
    id_anio_escolar INT AUTO_INCREMENT PRIMARY KEY,
    es_anio_actual TINYINT(1),
    fecha_inicio DATE,
    fecha_fin DATE
) ENGINE=InnoDB;

CREATE TABLE periodo (
    id_periodo INT AUTO_INCREMENT PRIMARY KEY,
    id_anio_escolar INT NOT NULL,
    orden_periodo INT NOT NULL,
    fecha_inicio DATE NOT NULL,
    fecha_fin DATE NOT NULL,
    FOREIGN KEY (id_anio_escolar) REFERENCES anio_escolar(id_anio_escolar),
    UNIQUE (id_anio_escolar, orden_periodo)
) ENGINE=InnoDB;


CREATE TABLE vinculacion_docente (
    id_anio_escolar INT NOT NULL,
    id_persona INT NOT NULL,
    fecha_vinculacion DATE,
    PRIMARY KEY (id_anio_escolar, id_persona),
    FOREIGN KEY (id_anio_escolar) REFERENCES anio_escolar(id_anio_escolar),
    FOREIGN KEY (id_persona) REFERENCES docente(id_persona)
) ENGINE=InnoDB;

-- Esto evita que exista dos veces el "Grupo A" en "Prejardín" para el mismo "Año 1"
ALTER TABLE grupo ADD UNIQUE INDEX uq_grupo_anio (id_grado, id_anio_escolar, nombre_grupo);


CREATE TABLE grupo (
    id_grupo INT AUTO_INCREMENT PRIMARY KEY,
    id_grado INT NOT NULL,
    id_docente_titular INT,
    id_anio_escolar INT NOT NULL,
    nombre_grupo VARCHAR(5) NOT NULL,
    FOREIGN KEY (id_grado) REFERENCES grado(id_grado),
    FOREIGN KEY (id_docente_titular) REFERENCES docente(id_persona),
    FOREIGN KEY (id_anio_escolar) REFERENCES anio_escolar(id_anio_escolar)
) ENGINE=InnoDB;

-- 2. La creamos de nuevo con la estructura que el código de Jhoan requiere
CREATE TABLE grupo_asignatura (
    id_grupo_asignatura INT AUTO_INCREMENT PRIMARY KEY,
    id_grupo INT NOT NULL,
    id_asignatura INT NOT NULL,
    id_docente INT DEFAULT NULL,
    -- Mantenemos la regla de que no se repita la misma materia en el mismo grupo
    UNIQUE KEY uq_grupo_asig (id_grupo, id_asignatura),
    FOREIGN KEY (id_grupo) REFERENCES grupo(id_grupo) ON DELETE CASCADE,
    FOREIGN KEY (id_asignatura) REFERENCES asignatura(id_asignatura) ON DELETE CASCADE,
    FOREIGN KEY (id_docente) REFERENCES docente(id_persona) ON DELETE SET NULL
) ENGINE=InnoDB;

CREATE TABLE matricula (
    id_matricula INT AUTO_INCREMENT PRIMARY KEY,
    id_estudiante INT NOT NULL,
    id_grupo INT NOT NULL,
    id_anio_escolar INT NOT NULL,
    fecha_matricula DATE NOT NULL,
    estado ENUM('Activa','Cancelada','Finalizada') DEFAULT 'Activa',
    FOREIGN KEY (id_estudiante) REFERENCES estudiante(id_persona),
    FOREIGN KEY (id_grupo) REFERENCES grupo(id_grupo),
    FOREIGN KEY (id_anio_escolar) REFERENCES anio_escolar(id_anio_escolar),
    UNIQUE (id_estudiante, id_anio_escolar)
) ENGINE=InnoDB;

CREATE TABLE desempenio_grupo_periodo (
    id_desempenio INT,
    id_grupo INT NOT NULL,
    id_periodo INT NOT NULL,
    FOREIGN KEY (id_desempenio) REFERENCES desempenio(id_desempenio),
    FOREIGN KEY (id_grupo) REFERENCES grupo (id_grupo),    
    FOREIGN KEY (id_periodo) REFERENCES periodo(id_periodo)
) ENGINE=InnoDB;
-- =========================
-- NOTAS Y CONDUCTA
-- =========================

CREATE TABLE nota (
    id_nota INT AUTO_INCREMENT PRIMARY KEY,
    id_matricula INT NOT NULL,
    id_asignatura INT NOT NULL,
    id_periodo INT NOT NULL,
    valor DECIMAL(4,2) NOT NULL,
    observacion_nota TEXT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_matricula) REFERENCES matricula(id_matricula),
    FOREIGN KEY (id_asignatura) REFERENCES asignatura(id_asignatura),
    FOREIGN KEY (id_periodo) REFERENCES periodo(id_periodo),
    UNIQUE (id_matricula, id_asignatura, id_periodo)
) ENGINE=InnoDB;

CREATE TABLE conducta (
    id_conducta INT AUTO_INCREMENT PRIMARY KEY,
    id_matricula INT NOT NULL,
    id_periodo INT NOT NULL,
    valor DECIMAL(4,2) NOT NULL,
    observacion_conducta TEXT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_matricula) REFERENCES matricula(id_matricula),
    FOREIGN KEY (id_periodo) REFERENCES periodo(id_periodo),
    UNIQUE (id_matricula, id_periodo)
) ENGINE=InnoDB;

SET FOREIGN_KEY_CHECKS = 1;
